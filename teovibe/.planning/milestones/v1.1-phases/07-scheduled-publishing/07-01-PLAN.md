       ```ruby
       class PublishPostJob < ApplicationJob
         queue_as :default
         discard_on ActiveJob::DeserializationError

         def perform(post_id)
           post = Post.find_by(id: post_id)
           return unless post&.scheduled?
           post.update!(status: :published, scheduled_at: nil, job_id: nil)
         end
       end
       ```
