1. Sortable.js 설치: `cd teovibe && pnpm add sortablejs`

2. `app/javascript/controllers/sortable_controller.js` 생성:
   ```javascript
   import { Controller } from "@hotwired/stimulus"
   import Sortable from "sortablejs"

   export default class extends Controller {
     static values = { url: String }

     connect() {
       this.sortable = Sortable.create(this.element, {
         animation: 150,
         handle: "[data-sortable-handle]",  // 드래그 핸들 (옵션)
         onEnd: this.onEnd.bind(this)
       })
     }

     disconnect() {
       this.sortable?.destroy()
     }

     onEnd() {
       const positions = [...this.element.querySelectorAll("[data-id]")].map(el => el.dataset.id)
       fetch(this.urlValue, {
         method: "PATCH",
         headers: {
           "Content-Type": "application/json",
           "X-CSRF-Token": document.querySelector('meta[name="csrf-token"]')?.content
         },
         body: JSON.stringify({ positions })
       })
     }
   }
   ```
